### 2.2 主循环 — `User/demo.c`

**说明:** 初始化所有外设 (ADC → PWM → MS901M → Modbus), 主循环中持续读取姿态传感器、更新寄存器、处理 Modbus 请求。

```c
// 文件: User/demo.c  行号: 34-109
void demo_run(void)
{
    // --- 初始化阶段 ---
    adc_init();                            // ADC 初始化 (5通道)
    pwm_init(19999, 83);                   // PWM 50Hz (10通道)
    atk_ms901m_init(115200);               // MS901M 姿态传感器
    modbus_init();                         // Modbus RTU 从站

    // --- 主循环 ---
    while (1) {
        atk_ms901m_attitude_data_t att;
        atk_ms901m_gyro_data_t gyro;

        // 读取姿态: Roll/Pitch/Yaw
        if (atk_ms901m_get_attitude(&att, 5) == ATK_MS901M_EOK) {
            modbus_set_register_float(REG_ROLL,  att.roll);
            modbus_set_register_float(REG_PITCH, att.pitch);
            modbus_set_register_float(REG_YAW,   att.yaw);
        }

        // 读取陀螺仪: Gyro X/Y/Z
        if (atk_ms901m_get_gyro_accelerometer(&gyro, NULL, 5) == ATK_MS901M_EOK) {
            modbus_set_register_float(REG_GYRO_X, gyro.x);
            modbus_set_register_float(REG_GYRO_Y, gyro.y);
            modbus_set_register_float(REG_GYRO_Z, gyro.z);
        }

        modbus_process();                  // 更新传感器 + 处理 Modbus 请求
        delay_ms(5);                       // 5ms 循环
    }
}
```

### 4.1 Modbus RTU 从站初始化 — `Drivers/BSP/Modbus/modbus.c`

**说明:** 基于 FreeModbus 初始化 Modbus RTU 从站 (地址 0x01, 9600 8N1)。

```c
// 文件: Drivers/BSP/Modbus/modbus.c  行号: 59-89
void modbus_init(void)
{
    // 清空寄存器
    for (int i = 0; i < REG_HOLDING_MAX; i++)
        g_modbus_registers[i] = 0;

    // 常量寄存器
    g_modbus_registers[REG_DEVICE_ID]  = 0x0407;
    g_modbus_registers[REG_FW_VERSION] = 0x0300;

    // 舵机默认中位 1500μs
    for (int i = REG_SERVO1; i <= REG_SERVO8; i++)
        g_modbus_registers[i] = 1500;

    // 初始化 Modbus RTU 从站
    eMBInit(MB_RTU, 0x01, 2, 9600, MB_PAR_NONE, 1);
    eMBEnable();
}
```

### 4.2 传感器数据更新到寄存器 — `Drivers/BSP/Modbus/modbus.c`

// 文件: Drivers/BSP/Modbus/modbus.c  行号: 91-114
void modbus_update_sensors(void)
{
    uint32_t tick = HAL_GetTick();
    g_modbus_registers[REG_SYS_TICK_L] = (uint16_t)(tick & 0xFFFF);
    g_modbus_registers[REG_SYS_TICK_H] = (uint16_t)(tick >> 16);

    // 4路模拟输入 (10次平均)
    g_modbus_registers[REG_ANALOG1] = adc_get_analog_value(ADC_CH_ANALOG1);
    g_modbus_registers[REG_ANALOG2] = adc_get_analog_value(ADC_CH_ANALOG2);
    g_modbus_registers[REG_ANALOG3] = adc_get_analog_value(ADC_CH_ANALOG3);
    g_modbus_registers[REG_ANALOG4] = adc_get_analog_value(ADC_CH_ANALOG4);

    // 电压检测 ×100
    g_modbus_registers[REG_VOLTAGE] = (uint16_t)(adc_get_power_voltage() * 100.0f);

    // ADC 原始值 (单次采样)
    g_modbus_registers[REG_ADC_RAW0] = adc_get_channel_value(ADC_CH_ANALOG1);
    g_modbus_registers[REG_ADC_RAW1] = adc_get_channel_value(ADC_CH_ANALOG2);
    g_modbus_registers[REG_ADC_RAW2] = adc_get_channel_value(ADC_CH_ANALOG3);
    g_modbus_registers[REG_ADC_RAW3] = adc_get_channel_value(ADC_CH_ANALOG4);
    g_modbus_registers[REG_ADC_RAW4] = adc_get_channel_value(ADC_CH_VOLTAGE);
}

### 4.3 Modbus 回调函数 (读写保持寄存器) — `Drivers/BSP/Modbus/modbus.c`

**说明:** FreeModbus 回调, 处理 FC03 读 / FC06 写 / FC10 批量写。含写保护和即时硬件执行。

```c
// 文件: Drivers/BSP/Modbus/modbus.c  行号: 171-210
eMBErrorCode eMBRegHoldingCB(UCHAR *pucRegBuffer, USHORT usAddress,
                              USHORT usNRegs, eMBRegisterMode eMode)
{
    usAddress--;  // FreeModbus 1-based → 0-based

    if ((usAddress + usNRegs) > REG_HOLDING_MAX)
        return MB_ENOREG;

    for (int i = 0; i < usNRegs; i++) {
        if (eMode == MB_REG_READ) {
            // 读: 高字节在前
            *pucRegBuffer++ = (UCHAR)(g_modbus_registers[usAddress] >> 8);
            *pucRegBuffer++ = (UCHAR)(g_modbus_registers[usAddress] & 0xFF);
        } else {
            // 写: 检查权限
            if (!is_reg_writable(usAddress))
                return MB_ENOREG;
            UCHAR hi = *pucRegBuffer++;
            UCHAR lo = *pucRegBuffer++;
            g_modbus_registers[usAddress] = (hi << 8) | lo;
            apply_register(usAddress);     // 立即执行 PWM 更新
        }
        usAddress++;
    }
    return MB_ENOERR;
}
```

### 4.4 IEEE 754 浮点寄存器存取 — `Drivers/BSP/Modbus/modbus.c`

```c
// 文件: Drivers/BSP/Modbus/modbus.c  行号: 144-167
void modbus_set_register_float(uint16_t addr, float value)
{
    uint32_t raw;
    memcpy(&raw, &value, sizeof(float));
    // Big-Endian 字序: 高字在低地址
    g_modbus_registers[addr]     = (uint16_t)(raw >> 16);
    g_modbus_registers[addr + 1] = (uint16_t)(raw & 0xFFFF);
}

float modbus_get_register_float(uint16_t addr)
{
    float value;
    uint32_t raw = ((uint32_t)g_modbus_registers[addr] << 16)
                 | g_modbus_registers[addr + 1];
    memcpy(&value, &raw, sizeof(float));
    return value;
}
```

### 5.1 上位机 Modbus CRC16 校验 — `modbus-dashboard/src/lib/modbus.ts`

```typescript
// 文件: modbus-dashboard/src/lib/modbus.ts  行号: 8-39
const CRC_TABLE = new Uint16Array(256);
for (let i = 0; i < 256; i++) {
  let crc = i;
  for (let j = 0; j < 8; j++) {
    crc = crc & 1 ? (crc >> 1) ^ 0xa001 : crc >> 1;
  }
  CRC_TABLE[i] = crc;
}

function crc16(data: Uint8Array): number {
  let crc = 0xffff;
  for (let i = 0; i < data.length; i++) {
    crc = (crc >> 8) ^ CRC_TABLE[(crc ^ data[i]) & 0xff];
  }
  return crc;
}
```

function buildReadHolding(slaveId: number, startAddr: number, numRegs: number): Uint8Array {
  const frame = new Uint8Array([

    (startAddr >> 8) & 0xff, startAddr & 0xff,

// Big-Endian 字序: 两个 16-bit 寄存器 → 32-bit float
function regsToFloat(hi: number, lo: number): number {
  const buf = new ArrayBuffer(4);
  const view = new DataView(buf);
  view.setUint16(0, hi, false);  // Big-Endian
  view.setUint16(2, lo, false);
  return view.getFloat32(0, false);
}

### 5.4 上位机事务队列 (防冲突) — `modbus-dashboard/src/lib/modbus.ts`

```typescript
// 异步互斥锁, 保证 Modbus 半双工总线同一时刻只有一个事务
private _queue: Array<() => void> = [];
private _busy = false;

private async _enqueue<T>(fn: () => Promise<T>): Promise<T> {
  if (this._busy) {
    await new Promise<void>(resolve => this._queue.push(resolve));
  }
  this._busy = true;
  try {
    return await fn();
  } finally {
    this._busy = false;
    this._queue.shift()?.();
  }
}
```
